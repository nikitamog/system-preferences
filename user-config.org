* =nm-init= startup behavior

#+begin_src emacs-lisp :comments no :tangle "nm-init.el"
;;; -*- lexical-binding: t -*-
#+end_src
trying out prot style:
1. split org documents into chunks of .el files for execution purposes
2. track only `this-user-config-file` and `init.el` for version control purposes
3. Generate files listed in literate config document.
4. Generate file, listing the build sequence as listed in document (since order matters.)
5. Now that config is prepped, load emacs init.
6. load build sequence file into a list
7. load files in the build sequence; continue on failure.

Dump build sequence to file.
#+begin_src emacs-lisp :tangle no :results none
  (defun nm/get-delimited-prefix (raw-string delimiter)
    (let ((regex (format "%s\\(.*?\\)%s" delimiter delimiter)))
      (if (string-match regex raw-string)
          (match-string 1 raw-string)
        nil)))

  (defun nm/build-order (buffer)
    (with-current-buffer buffer
      (unless (derived-mode-p 'org-mode)
        (user-error "Not in an org-mode buffer"))
      (let (ordered-list)
        (org-element-map (org-element-parse-buffer) 'headline
          (lambda (hl)
            (let ((header-title
                   (string-trim
                    (org-element-interpret-data (org-element-property :title hl)))))
              (push (format "%s.el" (nm/get-delimited-prefix header-title "=")) ordered-list))))
        (seq-uniq (nreverse ordered-list)))))

  (defun nm/dump-emacs-build-sequence ()
    (interactive)
    (let* ((target              "nm-emacs-build-sequence.txt")
           (buffer              (get-buffer-create target))
           (emacs-config-file "user-config.org"))
      (with-current-buffer buffer
        (erase-buffer)
        
        (insert (string-join (nm/build-order emacs-config-file) "\n"))
        (write-region (point-min) (point-max) target)
        (message "Done!"))))

  (defun nm/reload-emacs-config ()
    (interactive)
    (nm/dump-emacs-build-sequence)
    (with-current-buffer (find-file-noselect (locate-user-emacs-file "user-config.org"))
      (org-babel-tangle))
    (nm/load-nm-files "~/.emacs.d"))
#+end_src
